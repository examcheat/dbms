Q20)
(Stored Procedure using INOUT parameter: Table: Employee (ID, Name, Dept, Salary) Task: • Create a procedure that takes Dept as input through some parameter • Displays average salary of all employees of that department using same parameter as output.)
DELIMITER $$ 

CREATE PROCEDURE AvgSalaryByDept( 
INOUT DeptName VARCHAR(20) 
) 
BEGIN 
SELECT AVG(Salary) AS AverageSalary 
FROM Employee 
WHERE Dept = DeptName; 
END $$ 

DELIMITER ;
